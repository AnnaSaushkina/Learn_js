**Стрелочные функции внутри метода объекта не имеют своей области видимости this**

```js
let user = {
  name: 'Alice',
  greet1: function() {
    console.log(`Hello1, ${this.name}`)
  },
  greet2: () => {
    console.log(`Hello2, ${this.name}`)
  }
}

user.greet1() //"Hello1, Alice"
user.greet2() // "Hello2, undefined"
```
